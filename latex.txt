\section{\secmark{2}\; Guardian 4-Layer Defense-in-Depth Flow}
\label{sec:layers}
\vspace{-2mm}
{\small\itshape Sequential security pipeline with block conditions per layer}

\vspace{6mm}

\begin{figure}[H]
\centering
\begin{tikzpicture}[
  every node/.style={font=\sffamily\small},
  layernd/.style={
    draw=#1, line width=1.5pt, rounded corners=6pt,
    fill=#1!6, minimum width=14cm, align=left,

  },
  arrowdn/.style={-{Stealth[length=7pt]}, line width=1.5pt, color=accentcyan},
]

% Entry
\node[draw=darknavy, rounded corners=4pt, fill=darknavy!8,
      minimum width=8cm, align=center, inner sep=6pt]
  (entry) at (0,0) {\faUser\enspace User / Orchestrator sends input};

\draw[arrowdn] (entry.south) -- ++(0,-0.7);

% L1
\node[layernd=layerL1, below=1cm of entry]
  (l1box) {%
    \layertag{layerL1}{L1}\;\textbf{INPUT SHIELD}\\[6pt]
    \begin{minipage}{12.5cm}
    \begin{itemize}[leftmargin=1.2em, itemsep=2pt, parsep=0pt, topsep=0pt]
      \item Tier\,0:\;Exact blocklist match \hfill $<0.1$\,ms\enspace FREE
      \item Tier\,1:\;33-pattern regex (\texttt{SYSTEM OVERRIDE}, \texttt{curl},
            \texttt{/etc/passwd}\,\ldots) \hfill $<1$\,ms\enspace FREE
            $\;\to$ \textcolor{blockred}{\textbf{BLOCK}}
      \item Tier\,2:\;Azure AI Prompt Shields --- jailbreak + indirect injection
            \hfill $\sim$50\,ms $\;\to$ \textcolor{blockred}{\textbf{BLOCK}}
    \end{itemize}
    \end{minipage}
  };

\draw[arrowdn] (l1box.south) -- ++(0,-0.6)
  node[right, font=\sffamily\tiny, text=passgreen, xshift=3pt] {clean inputs only};

% Agent LLM
\node[draw=medgray, rounded corners=4pt, fill=medgray!15,
      minimum width=10cm, align=center, inner sep=5pt, below=0.9cm of l1box]
  (llm) {Agent LLM runs --- generates response + optional tool calls};

\draw[arrowdn] (llm.south) -- ++(0,-0.6);

% L4
\node[layernd=layerL4, below=0.9cm of llm]
  (l4box) {%
    \layertag{layerL4}{L4}\;\textbf{RBAC + BEHAVIORAL}%
    \enspace{\scriptsize(in-process, zero API cost)}\\[6pt]
    \begin{minipage}{12.5cm}
    \begin{itemize}[leftmargin=1.2em, itemsep=2pt, parsep=0pt, topsep=0pt]
      \item ABAC: role $\times$ verb $\times$ resource\_sensitivity $\times$ risk\_score
      \item Outcome: \textcolor{passgreen}{ALLOW} $|$
            \textcolor{blockred}{DENY $\to$ BLOCK} $|$
            \textcolor{warnambr}{ELEVATE $\to$ HITL queue}
      \item Behavioral signals:
            \ding{192}\,Z-score ($2.5\sigma$)\;
            \ding{193}\,Levenshtein ($0.4$)\;
            \ding{194}\,read$\to$exfil\;
            \ding{195}\,domain\;
            \ding{196}\,entropy ($1.5\times$)
    \end{itemize}
    \end{minipage}
  };

\draw[arrowdn] (l4box.south) -- ++(0,-0.6);

% L3
\node[layernd=layerL3, below=0.9cm of l4box]
  (l3box) {%
    \layertag{layerL3}{L3}\;\textbf{TOOL FIREWALL}\\[6pt]
    \begin{minipage}{12.5cm}
    \begin{itemize}[leftmargin=1.2em, itemsep=2pt, parsep=0pt, topsep=0pt]
      \item \texttt{file\_system}: path allowlist; blocks
            \texttt{.env .pem .key .ssh /etc/}
      \item \texttt{sql\_query}: SELECT only --- blocks DROP/DELETE/UPDATE/INSERT
      \item \texttt{shell\_cmds}: denylist
            \texttt{rm curl wget bash sh sudo eval exec}
      \item \texttt{http\_get/post}: HTTPS only; blocks 169.254.x.x IMDS + RFC-1918
      \item C1 Azure entity recog $\cdot$ C2 MELON (opt-in) $\cdot$ C4 HITL/AITL
    \end{itemize}
    \end{minipage}
  };

\draw[arrowdn] (l3box.south) -- ++(0,-0.6)
  node[right, font=\sffamily\tiny, text=passgreen, xshift=3pt]
  {tool executes only if all pass};

% Tool result
\node[draw=medgray, rounded corners=4pt, fill=medgray!15,
      minimum width=10cm, align=center, inner sep=5pt, below=0.9cm of l3box]
  (toolret) {Tool returns result $\to$ Agent generates final response};

\draw[arrowdn] (toolret.south) -- ++(0,-0.6);

% L2
\node[layernd=layerL2, below=0.9cm of toolret]
  (l2box) {%
    \layertag{layerL2}{L2}\;\textbf{OUTPUT GUARDRAIL}\\[6pt]
    \begin{minipage}{12.5cm}
    \begin{itemize}[leftmargin=1.2em, itemsep=2pt, parsep=0pt, topsep=0pt]
      \item Azure AI Language $\to$ PII redaction (SSN, API keys, email)
      \item Azure Content Safety $\to$ Toxicity classification + filter
      \item LLM-as-judge Groundedness $\to$ score 1--5; threshold 3
    \end{itemize}

\section{\secmark{10}\; Design Decisions Matrix}
\label{sec:decisions}
\vspace{-2mm}
{\small\itshape Key architectural choices, rationale, and explicit trade-offs}

\vspace{6mm}

\renewcommand{\arraystretch}{1.35}
\noindent
{\small
\begin{longtable}{%
  >{\bfseries\color{accentorange}\raggedright}p{2.5cm}
  >{\raggedright}p{2.8cm}
  >{\raggedright}p{5cm}
  >{\raggedright\arraybackslash}p{4.5cm}}
\toprule
\textbf{Decision} & \textbf{Choice Made} & \textbf{Rationale}
  & \textbf{Trade-off / Deferred Alt.} \\
\midrule
\endfirsthead
\toprule
\textbf{Decision} & \textbf{Choice Made} & \textbf{Rationale}
  & \textbf{Trade-off / Deferred Alt.} \\
\midrule
\endhead

L2 provider &
  Azure AI Language + Content Safety &
  Single trust boundary with L1; avoids second external model dependency &
  Guardrails AI deferred as optional future plugin \\
\addlinespace

Tiered L1 (offline first) &
  Tier 0/1 regex before Azure API call &
  $\sim$40--60\,\% blocked free at $<$1\,ms;
  Azure only for ambiguous inputs &
  Regex misses novel patterns $\to$ Tier 2 catches those \\
\addlinespace

ABAC ELEVATE outcome &
  3rd outcome: ELEVATE $\to$ HITL queue &
  Binary ALLOW/DENY blocks valid edge-cases;
  ELEVATE routes to human without dropping the task &
  Adds async complexity; requires HITL UI / webhook \\
\addlinespace

C2 MELON opt-in &
  Disabled by default; enabled in YAML &
  Requires 2 LLM calls (2$\times$ cost); only justified for
  medical/financial/legal agents &
  Always-on = 2$\times$ cost; config flag is the balance \\
\addlinespace

Dual observability (OTel + SQLite) &
  OTel for live SRE; SQLite for forensics &
  OTel spans are ephemeral; compliance audits need local record
  independent of cloud connectivity &
  Two writes per event; acceptable overhead \\
\addlinespace

Redis caching &
  Config-stubbed (not wired yet) &
  Architecture validated; wiring deferred to avoid prototype
  complexity &
  Hot path is $\sim$200\,ms instead of target $<$50\,ms \\
\addlinespace

Spotlighting &
  Azure AI Foundry endpoint required &
  Best-in-class indirect injection defence for
  document-grounded agents &
  Not available in prototype; falls back to Tier-1 regex \\
